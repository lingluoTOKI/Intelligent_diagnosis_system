```python
# 搜索功能
def search_chat_history(self, keyword):
    """搜索对话历史"""
    results = []
    for i, chat in enumerate(self.chat_history):
        if keyword.lower() in chat['question'].lower() or keyword.lower() in chat['answer'].lower():
            results.append({
                'index': i + 1,
                'question': chat['question'],
                'answer': chat['answer'],
                'timestamp': chat['timestamp']
            })
    return results
```
